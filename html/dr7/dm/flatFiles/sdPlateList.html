typedef struct {

	int plateId;
	double ra;			# RA at center of plate
	double dec;			# DEC at center of plate
	float airmax;			# maximum airmass can be done at
	char recvStat[20];		# plate at drillers or received on mountain or defective
	char markStat[20];		# plate marked (important holes labeled)	
	char stuffStat[20]; 		# is plate currently stuffed?

	int stuffPriority;		# stuff priority, 0=low, 1=higher

} SDPLATELIST;
